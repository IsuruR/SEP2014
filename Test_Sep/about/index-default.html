Video gallery - from API

Import a group of videos from youtube API, set fallback URL for accessibility, SEO etc...

$(document).ready(function(){
    $("ul.youtube-videogallery2").youtubeVideoGallery( {
        apiFallbackUrl: 'https://www.youtube.com/results?search_query=dance+tango',
        apiUrl: 'https://gdata.youtube.com/feeds/api/videos?q=dance+tango&orderby=published&start-index=1&alt=json&max-results=10&v=2'
     });
});
